> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `trips` Django app with `Trip` and `TripStop` models, a nested serializer, and user-scoped CRUD endpoints at `/api/trips/`.

**Architecture:** Standard ModelViewSet + DefaultRouter pattern matching the existing `people` and `stations` apps. `TripSerializer` embeds stops using a nested `TripStopSerializer`; on write, stops are deleted and recreated to handle reordering. All endpoints require authentication; trips are filtered to `request.user`.

**Tech Stack:** Django/GeoDjango, DRF, `rest_framework_gis`, PostgreSQL/PostGIS, Docker

from django.conf import settings
from django.contrib.gis.db import models


class Trip(models.Model):
    name = models.CharField(max_length=255)
    date = models.DateField()
    user = models.ForeignKey(
        settings.AUTH_USER_MODEL,
        on_delete=models.CASCADE,
        related_name='trips',
    )

    def __str__(self):
        return f"{self.name} ({self.date})"


class TripStop(models.Model):
    trip = models.ForeignKey(Trip, on_delete=models.CASCADE, related_name='stops')
    person = models.ForeignKey('people.Person', on_delete=models.CASCADE)
    sequence_order = models.PositiveIntegerField()
    location = models.PointField()

    class Meta:
        ordering = ['sequence_order']

    def __str__(self):
        return f"Stop {self.sequence_order} on {self.trip}"
```

from rest_framework import serializers
from rest_framework_gis.fields import GeometryField
from .models import Trip, TripStop


class TripStopSerializer(serializers.ModelSerializer):
    location = GeometryField()

    class Meta:
        model = TripStop
        fields = ('id', 'person', 'sequence_order', 'location')


class TripSerializer(serializers.ModelSerializer):
    stops = TripStopSerializer(many=True)

    class Meta:
        model = Trip
        fields = ('id', 'name', 'date', 'stops')

    def create(self, validated_data):
        stops_data = validated_data.pop('stops')
        trip = Trip.objects.create(**validated_data)
        for stop_data in stops_data:
            TripStop.objects.create(trip=trip, **stop_data)
        return trip

    def update(self, instance, validated_data):
        stops_data = validated_data.pop('stops')
        instance.name = validated_data.get('name', instance.name)
        instance.date = validated_data.get('date', instance.date)
        instance.save()
        instance.stops.all().delete()
        for stop_data in stops_data:
            TripStop.objects.create(trip=instance, **stop_data)
        return instance
```
